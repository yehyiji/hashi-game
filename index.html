    function HashiGame() {

      const [currentLevelId, setCurrentLevelId] = useState(1);

      const [bridges, setBridges] = useState([]);

      const [history, setHistory] = useState([]); 

      const [isWon, setIsWon] = useState(false);

      const [message, setMessage] = useState("");


      const [isMenuOpen, setIsMenuOpen] = useState(window.innerWidth > 768);

      const [scale, setScale] = useState(1);

      const containerRef = useRef(null);

      const scaleRef = useRef(1);


      useEffect(() => { scaleRef.current = scale; }, [scale]);


      const currentLevel = useMemo(() => LEVELS.find(l => l.id === currentLevelId), [currentLevelId]);


      const potentialBridges = useMemo(() => {

        let pb = [];

        let bridgeId = 1;

        const islands = currentLevel.islands;


        for (let i = 0; i < islands.length; i++) {

          let a = islands[i];

          

          let rightNeighbor = null;

          for (let j = 0; j < islands.length; j++) {

            let b = islands[j];

            if (b.y === a.y && b.x > a.x) {

              if (!rightNeighbor || b.x < rightNeighbor.x) {

                rightNeighbor = b;

              }

            }

          }

          if (rightNeighbor) {

            pb.push({ id: bridgeId++, from: a.id, to: rightNeighbor.id, type: 'h', count: 0 });

          }


          let bottomNeighbor = null;

          for (let j = 0; j < islands.length; j++) {

            let b = islands[j];

            if (b.x === a.x && b.y > a.y) {

              if (!bottomNeighbor || b.y < bottomNeighbor.y) {

                bottomNeighbor = b;

              }

            }

          }

          if (bottomNeighbor) {

            pb.push({ id: bridgeId++, from: a.id, to: bottomNeighbor.id, type: 'v', count: 0 });

          }

        }

        return pb;

      }, [currentLevel]);


      useEffect(() => {

        setBridges(potentialBridges.map(b => ({ ...b, count: 0 })));

        setHistory([]); 

        setIsWon(false);

        setMessage("");

        setScale(1); 

        if (window.innerWidth <= 768) {

          setIsMenuOpen(false); 

        }

      }, [potentialBridges, currentLevelId]);


      useEffect(() => {

        const container = containerRef.current;

        if (!container) return;


        let initialDist = 0;

        let initialScale = 1;


        const handleTouchStart = (e) => {

          if (e.touches.length === 2) {

            const dx = e.touches[0].clientX - e.touches[1].clientX;

            const dy = e.touches[0].clientY - e.touches[1].clientY;

            initialDist = Math.hypot(dx, dy);

            initialScale = scaleRef.current;

          }

        };


        const handleTouchMove = (e) => {

          if (e.touches.length === 2) {

            e.preventDefault(); 

            const dx = e.touches[0].clientX - e.touches[1].clientX;

            const dy = e.touches[0].clientY - e.touches[1].clientY;

            const dist = Math.hypot(dx, dy);

            const newScale = Math.min(Math.max(0.3, initialScale * (dist / initialDist)), 3.5); 

            setScale(newScale);

          }

        };


        container.addEventListener('touchstart', handleTouchStart, { passive: false });

        container.addEventListener('touchmove', handleTouchMove, { passive: false });


        return () => {

          container.removeEventListener('touchstart', handleTouchStart);

          container.removeEventListener('touchmove', handleTouchMove);

        };

      }, []);


      const islandCurrentCounts = useMemo(() => {

        const counts = {};

        currentLevel.islands.forEach(i => counts[i.id] = 0);

        bridges.forEach(b => {

          if (counts[b.from] !== undefined) counts[b.from] += b.count;

          if (counts[b.to] !== undefined) counts[b.to] += b.count;

        });

        return counts;

      }, [bridges, currentLevel]);


      useEffect(() => {

        let allCountsMatch = true;

        for (let island of currentLevel.islands) {

          if (islandCurrentCounts[island.id] !== island.req) {

            allCountsMatch = false;

            break;

          }

        }


        if (!allCountsMatch) {

          setIsWon(false);

          return;

        }


        const adjList = {};

        currentLevel.islands.forEach(i => adjList[i.id] = []);

        bridges.forEach(b => {

          if (b.count > 0 && adjList[b.from] && adjList[b.to]) {

            adjList[b.from].push(b.to);

            adjList[b.to].push(b.from);

          }

        });


        const visited = new Set();

        const queue = [currentLevel.islands[0].id];

        if(queue.length > 0) visited.add(currentLevel.islands[0].id);


        while (queue.length > 0) {

          const curr = queue.shift();

          for (let neighbor of adjList[curr]) {

            if (!visited.has(neighbor)) {

              visited.add(neighbor);

              queue.push(neighbor);

            }

          }

        }


        if (visited.size === currentLevel.islands.length) {

          setIsWon(true);

        } else {

          setIsWon(false);

        }

      }, [bridges, currentLevel, islandCurrentCounts]);


      const showMessage = (msg) => {

        setMessage(msg);

        setTimeout(() => setMessage(""), 3000);

      };


      const isCrossing = (targetBridge, currentBridges) => {

        const islandsMap = {};

        currentLevel.islands.forEach(i => islandsMap[i.id] = i);


        const tFrom = islandsMap[targetBridge.from];

        const tTo = islandsMap[targetBridge.to];


        if (!tFrom || !tTo) return false;


        if (targetBridge.type === 'h') {

          const y = tFrom.y;

          const minX = Math.min(tFrom.x, tTo.x);

          const maxX = Math.max(tFrom.x, tTo.x);


          for (let b of currentBridges) {

            if (b.count > 0 && b.type === 'v') {

              const bFrom = islandsMap[b.from];

              const bTo = islandsMap[b.to];

              if (!bFrom || !bTo) continue;

              

              const x = bFrom.x;

              const minY = Math.min(bFrom.y, bTo.y);

              const maxY = Math.max(bFrom.y, bTo.y);


              if (x > minX && x < maxX && y > minY && y < maxY) {

                return true;

              }

            }

          }

        } else {

          const x = tFrom.x;

          const minY = Math.min(tFrom.y, tTo.y);

          const maxY = Math.max(tFrom.y, tTo.y);


          for (let b of currentBridges) {

            if (b.count > 0 && b.type === 'h') {

              const bFrom = islandsMap[b.from];

              const bTo = islandsMap[b.to];

              if (!bFrom || !bTo) continue;


              const y = bFrom.y;

              const minX = Math.min(bFrom.x, bTo.x);

              const maxX = Math.max(bFrom.x, bTo.x);


              if (y > minY && y < maxY && x > minX && x < maxX) {

                return true;

              }

            }

          }

        }

        return false;

      };


      const handleBridgeClick = (bridgeId) => {

        if (isWon) return; 


        setBridges(prev => {

          setHistory(past => [...past, prev]);


          return prev.map(b => {

            if (b.id !== bridgeId) return b;

            

            if (b.count === 0 && isCrossing(b, prev)) {

              showMessage("🚫 橋樑不能交叉喔！請先移除阻擋的橋。");

              setHistory(past => past.slice(0, -1));

              return b;

            }

            

            return { ...b, count: (b.count + 1) % 3 };

          });

        });

      };


      const handleUndo = () => {

        if (history.length === 0 || isWon) return;

        const previousState = history[history.length - 1];

        setBridges(previousState);

        setHistory(past => past.slice(0, -1));

      };


      const handleReset = () => {

        setBridges(potentialBridges.map(b => ({...b, count: 0})));

        setHistory([]);

      };


      const maxGridX = Math.max(...currentLevel.islands.map(i => i.x), 6); 

      const maxGridY = Math.max(...currentLevel.islands.map(i => i.y), 6);

      

      const SCALE = 65;

      const PADDING = 85;

      const ISLAND_RADIUS = 20;


      const baseSvgWidth = maxGridX * SCALE + PADDING * 2;

      const baseSvgHeight = maxGridY * SCALE + PADDING * 2;


      const renderBridge = (bridge) => {

        const islandsMap = {};

        currentLevel.islands.forEach(i => islandsMap[i.id] = i);

        const from = islandsMap[bridge.from];

        const to = islandsMap[bridge.to];


        if (!from || !to) return null;


        const x1 = from.x * SCALE + PADDING;

        const y1 = from.y * SCALE + PADDING;

        const x2 = to.x * SCALE + PADDING;

        const y2 = to.y * SCALE + PADDING;


        return (

          <g key={bridge.id} className="group cursor-pointer" onClick={() => handleBridgeClick(bridge.id)}>

            {bridge.type === 'h' ? (

              <rect

                x={Math.min(x1, x2) + ISLAND_RADIUS}

                y={y1 - 20}

                width={Math.abs(x1 - x2) - ISLAND_RADIUS * 2}

                height={40}

                className="fill-transparent group-hover:fill-blue-100/50 transition-colors"

              />

            ) : (

              <rect

                x={x1 - 20}

                y={Math.min(y1, y2) + ISLAND_RADIUS}

                width={40}

                height={Math.abs(y1 - y2) - ISLAND_RADIUS * 2}

                className="fill-transparent group-hover:fill-blue-100/50 transition-colors"

              />

            )}


            {bridge.count === 1 && (

              <line x1={x1} y1={y1} x2={x2} y2={y2} stroke="#374151" strokeWidth="6" />

            )}

            {bridge.count === 2 && bridge.type === 'h' && (

              <React.Fragment>

                <line x1={x1} y1={y1 - 5} x2={x2} y2={y2 - 5} stroke="#374151" strokeWidth="5" />

                <line x1={x1} y1={y1 + 5} x2={x2} y2={y2 + 5} stroke="#374151" strokeWidth="5" />

              </React.Fragment>

            )}

            {bridge.count === 2 && bridge.type === 'v' && (

              <React.Fragment>

                <line x1={x1 - 5} y1={y1} x2={x2 - 5} y2={y2} stroke="#374151" strokeWidth="5" />

                <line x1={x1 + 5} y1={y1} x2={x2 + 5} y2={y2} stroke="#374151" strokeWidth="5" />

              </React.Fragment>

            )}

          </g>

        );

      };


      const renderIsland = (island) => {

        const cx = island.x * SCALE + PADDING;

        const cy = island.y * SCALE + PADDING;

        const current = islandCurrentCounts[island.id];

        

        let strokeColor = "#1f2937";

        let bgColor = "#ffffff";

        let textColor = "#1f2937";


        if (current === island.req) {

          strokeColor = "#10b981"; // 綠色

          bgColor = "#ecfdf5";

          textColor = "#047857";

        } else if (current > island.req) {

          strokeColor = "#ef4444"; // 紅色

          bgColor = "#fef2f2";

          textColor = "#b91c1c";

        }


        return (

          <g key={`island-${island.id}`}>

            <circle

              cx={cx}

              cy={cy}

              r={ISLAND_RADIUS}

              fill={bgColor}

              stroke={strokeColor}

              strokeWidth="4"

              className="transition-colors duration-300"

            />

            <text

              x={cx}

              y={cy}

              textAnchor="middle"

              dominantBaseline="central"

              className="text-xl font-bold font-sans pointer-events-none transition-colors duration-300"

              fill={textColor}

            >

              {island.req}

            </text>

          </g>

        );

      };


      return (

        <div className="min-h-screen p-2 md:p-4 flex flex-col items-center font-sans">

          <div className="w-full max-w-5xl bg-white rounded-2xl shadow-xl overflow-hidden flex flex-col h-[95vh] md:max-h-[90vh] border border-slate-200">

            

            <div className="bg-blue-600 text-white shrink-0 z-20 shadow-md">

              <div className="p-3 md:p-4 flex flex-wrap justify-between items-center gap-2">

                <div className="flex items-center gap-3">

                  <h1 className="text-xl md:text-2xl font-bold">🌉 數橋</h1>

                  <span className="bg-blue-800/50 px-2 py-1 rounded text-xs md:text-sm truncate max-w-[120px] md:max-w-none">{currentLevel.name}</span>

                </div>

                

                <div className="flex items-center gap-1 md:gap-2 ml-auto">

                  <button 

                    onClick={handleUndo}

                    disabled={history.length === 0 || isWon}

                    className={`px-3 py-1.5 text-sm rounded-lg font-medium transition-colors flex items-center gap-1

                      ${history.length === 0 || isWon 

                        ? "bg-blue-800/40 text-blue-300 cursor-not-allowed" 

                        : "bg-amber-500 hover:bg-amber-600 text-white shadow-sm"}`}

                  >

                    ↩️ <span className="hidden md:inline">上一步</span>

                  </button>

                  <button 

                    onClick={handleReset}

                    className="px-3 py-1.5 bg-red-500 hover:bg-red-600 text-white text-sm rounded-lg font-medium transition-colors shadow-sm"

                  >

                    🔄 <span className="hidden md:inline">重置</span>

                  </button>

                  <button 

                    onClick={() => setIsMenuOpen(!isMenuOpen)}

                    className="px-3 py-1.5 bg-blue-700 hover:bg-blue-800 text-white text-sm rounded-lg font-medium transition-colors shadow-sm flex items-center gap-1"

                  >

                    {isMenuOpen ? '▲ 收起' : '▼ 選單'}

                  </button>

                </div>

              </div>


              {isMenuOpen && (

                <div className="p-3 md:p-4 pt-0 border-t border-blue-500/50 bg-blue-600">

                  <p className="text-blue-100 opacity-90 mb-3 text-sm">{currentLevel.description}</p>

                  <div className="flex flex-wrap gap-2 items-center">

                    {LEVELS.map(level => (

                      <button

                        key={level.id}

                        onClick={() => setCurrentLevelId(level.id)}

                        className={`px-3 py-1.5 rounded-lg text-sm font-medium transition-all ${

                          currentLevelId === level.id 

                            ? "bg-white text-blue-600 shadow-sm" 

                            : "bg-blue-700/50 hover:bg-blue-700 text-white"

                        }`}

                      >

                        {level.name}

                      </button>

                    ))}

                  </div>

                </div>

              )}

            </div>


            <div className="h-8 md:h-10 shrink-0 flex items-center justify-center bg-slate-100 border-b border-slate-200 z-10 shadow-sm">

              {message && <span className="text-red-600 font-semibold animate-pulse text-sm">{message}</span>}

              {!message && !isWon && <span className="text-slate-500 text-xs md:text-sm">操作提示：兩指捏放可縮放，點空白處切換橋樑。</span>}

              {isWon && <span className="text-emerald-600 font-bold text-sm md:text-lg animate-bounce">🎉 恭喜！完美過關！ 🎉</span>}

            </div>


            <div 

              ref={containerRef}

              className="p-4 bg-slate-200/50 flex-1 overflow-auto relative custom-scrollbar touch-pan-x touch-pan-y"

              style={{ overscrollBehavior: 'none' }}

            >

              <div className="flex justify-center items-center min-w-max min-h-max">

                <svg 

                  viewBox={`0 0 ${baseSvgWidth} ${baseSvgHeight}`} 

                  className="bg-white rounded-xl shadow-inner transition-all duration-75 origin-center"

                  style={{ 

                    width: `${baseSvgWidth * scale}px`,

                    height: `${baseSvgHeight * scale}px`,

                    backgroundImage: 'radial-gradient(#e2e8f0 2px, transparent 2px)',

                    backgroundSize: '40px 40px' 

                  }}

                >

                  {bridges.map(renderBridge)}

                  {currentLevel.islands.map(renderIsland)}

                </svg>

              </div>


              <div className="fixed md:absolute bottom-6 right-6 flex flex-col gap-2 z-20">

                <button onClick={() => setScale(s => Math.min(s + 0.3, 3.5))} className="w-10 h-10 bg-white/90 backdrop-blur rounded-full shadow-lg flex items-center justify-center text-2xl font-bold text-slate-700 border border-slate-200 hover:bg-slate-50 active:scale-95 transition-transform">＋</button>

                <button onClick={() => setScale(s => Math.max(s - 0.3, 0.3))} className="w-10 h-10 bg-white/90 backdrop-blur rounded-full shadow-lg flex items-center justify-center text-2xl font-bold text-slate-700 border border-slate-200 hover:bg-slate-50 active:scale-95 transition-transform">－</button>

              </div>


              {isWon && (

                <div className="absolute inset-0 bg-white/60 backdrop-blur-sm flex items-center justify-center z-10">

                  <div className="bg-white p-8 rounded-2xl shadow-2xl text-center transform hover:scale-105 transition-transform border-2 border-emerald-400">

                    <div className="text-6xl mb-4">🏆</div>

                    <h2 className="text-3xl font-bold text-slate-800 mb-2">大師級工程師！</h2>

                    <p className="text-slate-600 mb-6">你成功連通了所有的島嶼。</p>

                    {currentLevelId < LEVELS.length ? (

                      <button 

                        onClick={() => setCurrentLevelId(currentLevelId + 1)}

                        className="px-6 py-3 bg-emerald-500 hover:bg-emerald-600 text-white text-lg font-bold rounded-xl shadow-lg hover:shadow-xl transition-all"

                      >

                        挑戰下一關 ➔

                      </button>

                    ) : (

                      <button 

                        onClick={handleReset}

                        className="px-6 py-3 bg-blue-500 hover:bg-blue-600 text-white text-lg font-bold rounded-xl shadow-lg hover:shadow-xl transition-all"

                      >

                        再玩一次

                      </button>

                    )}

                  </div>

                </div>

              )}

            </div>


          </div>

        </div>

      );

    }


    // 將 React 元件渲染到畫面上

    const root = ReactDOM.createRoot(document.getElementById('root'));

    root.render(<HashiGame />);

  </script>
